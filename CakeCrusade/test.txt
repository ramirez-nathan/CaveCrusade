::testing::AssertionResult IsBetweenInclusive(int val, int lowBound, int highBound) {
	if ((val >= lowBound) && (val <= highBound))
		return::testing::AssertionSuccess();
	else
		testing::AssertionFailure()
		<< val << " is outside the bounds of the map which are from " << lowBound << " to " << highBound;
}

TEST(TileMapTests, OnTheMap) {
	int a = 6;
	EXPECT_TRUE(IsBetweenInclusive(a, 0, 12));
}

TEST(TileMapTests, OnTheMap) {
	int a = 15;
	EXPECT_FALSE(IsBetweenInclusive(a, 0, 12));
}
